# useRControl

Hook to add React Control inside MapLibre

## Usage

```tsx
import { RMap, useRControl } from "maplibre-react-components";

function CustomControl() {
  const { container } = useRControl({
    position: "bottom-left"
  });

  return createPortal(
    <>
      <h3>Custom control</h3>
      <div>add your content</div>
    </>,
    container
  );
}

function App() {
  return (
    <RMap>
      <CustomControl />
    </RMap>
  );
}
```
