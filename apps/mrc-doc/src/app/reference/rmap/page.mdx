# RMap

Base component

## Usage

```tsx
import { RMap } from "maplibre-react-components";

const home = { lng: 6, lat: 46 };

export default function App() {

  return (
    <RMap
      initialCenter={home}
      style={{ width: "100%", height: "100%" }}
      mapStyle="https://demotiles.maplibre.org/style.json"
    />
  );
}
```

type RMapProps =
  /**
   * options for the class that controls the map instance, including
   * behavior when changing styles
   */
  ManagerOptions &

  /**
   * props of the <div> wrapper component
   */
  RMapComponentProps &

  /* props mapped to maplibre MapOptions*/
  MapProps;


type ManagerOptions = {
  /* si l'on propose un objet attention à donner un objet stable pour des raisons de performances la vérification se fait avec un === et non un deepEqual */
  mapStyle?: StyleSpecification | string;
  styleDiffing?: boolean;
  /* idem proposer un objet stable */
  styleTransformStyle?: TransformStyleFunction;
  padding?: PaddingOptions;
}

type RMapComponentProps = {
  children?: ReactNode;
  style?: CSSProperties;
  id?: string;
  className?: string;
  onMounted?: (map: Map) => void;
}


type MapProps =
  /**
   * Reactive map options. Update during the RMap lifecycle causes them to be
   * updated on the map instance
   * they have the same name as in MapOptions
   */
  MapReactiveOptions &

  /**
   * Reactive map handler options. Update during the RMap lifecycle causes handler
   * to be enabled/disabled
   * they have the same name as in MapOptions
   */
  MapHandlerOptions &

  /**
   * Listenners for map events
   * they have the same name as in MapOptions with first letter uppercase and prefixed
   * with "on"
   */
  MapCallbacks &

  /**
   * Non reactive map options that are only used when instantiating a map object
   * they have the same name as in MapOptions with first letter uppercase
   * and prefixed with "initial"
   */
  MapInitialOptions;
