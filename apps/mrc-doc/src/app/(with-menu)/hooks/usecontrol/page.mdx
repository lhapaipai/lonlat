# useControl

Hook to add Vanilla MapLibre Control inside RMap.

## Usage

```tsx
import { ControlPosition, IControl, LogoControl, LogoControlOptions } from "maplibre-gl";
import { memo } from "react";
import { useControl } from "maplibre-react-components";

type RLogoControlProps = LogoControlOptions & {
  position?: ControlPosition;
};

const RLogoControl = memo(function RLogoControl({ position = "bottom-left", ...controlOptions }: RLogoControlProps) {
  const control = useControl({
    position,
    factory: () => new LogoControl(controlOptions),
  });
  return null;
})


function App() {
  return (
    <RMap>
      <RLogoControl position="top-left" />
    </RMap>
  )
}
```
